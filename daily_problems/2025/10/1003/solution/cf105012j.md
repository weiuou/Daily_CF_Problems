**提示 1：** 很典型的给出一个路径权值要求最大化的题。

**提示 2：** 路径分为两部分，向上走，再往下走（从根的角度来看）。

很典型的换根 DP 问题。

这里给出的函数不具有结合律，所以只能使用换根。

最大化路径权值，主要就是把路径拆成向上走和向下走的两部分。

我们先处理向上走的部分，每次相当于在原有路径后面新增一个元素，因此是很容易实现子节点向父节点的转移的。同时 $f(x_1,x_2)$ 是关于 $x_1,x_2$ 都单调不减的，所以取最大的子节点路径权值是可以的。

接下来考虑往下走的部分。这条路径可能来自于父节点，也可能来自于兄弟节点。我们只需取最大的兄弟节点权值 / 取父节点对应的权值往下走就行。

时间复杂度为 $\mathcal{O}(n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    nums = LII()
    
    path = [[] for _ in range(n)]
    
    for _ in range(n - 1):
        u, v = GMI()
        path[u].append(v)
        path[v].append(u)
    
    parent = [-1] * n
    que = [0]
    
    for u in que:
        for v in path[u]:
            if parent[u] != v:
                parent[v] = u
                que.append(v)
    
    def f(x, y):
        return fmax(x, y) + math.isqrt(fmin(x, y))
    
    que.reverse()
    
    dp1 = [0] * n
    dp2 = [0] * n
    
    for u in que:
        dp1[u] = nums[u]
        
        for v in path[u]:
            if parent[v] == u:
                dp1[u] = fmax(dp1[u], f(dp1[v], nums[u]))
    
    que.reverse()
    
    for u in que:
        dp2[u] = fmax(dp2[u], nums[u])
        v1, v2 = dp2[u], 0
        
        for v in path[u]:
            if parent[v] == u:
                nv = f(dp1[v], nums[u])
                if nv > v1: v1, v2 = nv, v1
                elif nv > v2: v2 = nv
        
        for v in path[u]:
            if parent[v] == u:
                nv = f(dp1[v], nums[u])
                if nv == v1: dp2[v] = f(v2, nums[v])
                else: dp2[v] = f(v1, nums[v])
    
    print(fmax(max(dp1), max(dp2)))
```

C++ 做法如下——

```cpp []
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    int n;
    cin >> n;

    vector<long long> nums(n);
    for (auto &v: nums) cin >> v;

    vector<vector<int>> path(n);
    for (int i = 0; i < n - 1; i ++) {
        int u, v;
        cin >> u >> v;
        u --, v --;
        path[u].emplace_back(v);
        path[v].emplace_back(u);
    }

    vector<long long> dp1(n, 0), dp2(n, 0);

    auto f = [&] (long long x, long long y) -> long long {
        int v = sqrt(min(x, y));
        while (v * v < min(x, y)) v ++;
        while (v * v > min(x, y)) v --;
        return max(x, y) + v;
    };

    auto dfs1 = [&] (auto &self, int u, int p) -> void {
        dp1[u] = nums[u];
        for (auto &v: path[u]) {
            if (v != p) {
                self(self, v, u);
                dp1[u] = max(dp1[u], f(dp1[v], nums[u]));
            }
        }
    };

    dfs1(dfs1, 0, -1);

    auto dfs2 = [&] (auto &self, int u, int p) -> void {
        dp2[u] = max(dp2[u], nums[u]);

        long long v1 = dp2[u], v2 = 0;

        for (auto &v: path[u]) {
            if (v != p) {
                long long nv = f(dp1[v], nums[u]);
                if (nv > v1) swap(v1, v2), v1 = nv;
                else if (nv > v2) v2 = nv;
            }
        }

        for (auto &v: path[u]) {
            if (v != p) {
                long long nv = f(dp1[v], nums[u]);
                if (nv == v1) dp2[v] = f(v2, nums[v]);
                else dp2[v] = f(v1, nums[v]);
                self(self, v, u);
            }
        }
    };

    dfs2(dfs2, 0, -1);

    cout << max(*max_element(dp1.begin(), dp1.end()), *max_element(dp2.begin(), dp2.end()));

    return 0;
}
```
